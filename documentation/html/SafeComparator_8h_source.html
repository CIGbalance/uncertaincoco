48  //For overlap, potentially domineering one needs to be sent first
49  template <class Individual>
50  bool compSafe(const Individual & A, const Individual& B,
51  SelectionStrategy strat=NONE) {
52  if(strat==NONE){
53  strat=this->strat;
54  }
55  switch(strat){
56  case NO_UNCERTAINTY:
57  if(Individual::IsUncertain(A) || Individual::IsUncertain(B)){
58  return false;
59  }else{
60  return true;
61  }
62  case PREDICTION:
63  return true;
64  case NO_OVERLAP:
65  return !overlapChecker(A,B);
66  case MO:
67  throw std::invalid_argument("MO comparator not implemented");
68  case NONE:
69  throw std::invalid_argument("No comparison strategy given (none)");
70  }
71  throw std::invalid_argument("Comparison strategy unknown");
72  }

82  int operator()( const UncertainIndividual<double>& A, const UncertainIndividual<double>& B){
83  Extractor e;
84  int val = pdc.TRADE_OFF;
85  if(e(A) < e(B)){
86  val = pdc.A_STRICTLY_DOMINATES_B;
87  }else if(e(A) == e(B)){
88  val = pdc.A_EQUALS_B;
89  }else if(e(A) > e(B)){
90  val = pdc.B_STRICTLY_DOMINATES_A;
91  }
92  return checkUncertainty(A, B, val);
93  }

97  template <class Individual>
98  int checkUncertainty(const Individual& A, const Individual& B,
99  int priorRanking){
100  bool safe = false;
101  if(priorRanking==pdc.A_STRICTLY_DOMINATES_B || priorRanking==pdc.A_WEAKLY_DOMINATES_B){
102  safe = compSafe(A, B);
103  }else if(priorRanking==pdc.A_EQUALS_B){
104  if(this->strat!=PREDICTION){
105  safe = compSafe(A,B,NO_UNCERTAINTY);
106  }
107  }else if(priorRanking==pdc.B_STRICTLY_DOMINATES_A || priorRanking== pdc.B_WEAKLY_DOMINATES_A){
108  safe = compSafe(B, A);
109  }
110  if(safe){
111  return priorRanking;
112  }else{
113  return pdc.TRADE_OFF;
114  }

117  bool overlapChecker(const UncertainIndividual<double> & A, const UncertainIndividual<double> & B){
118  Extractor e;
119  double qA;
120  if(A.uncertainty()[0]>0){
121  normal a(e(A), A.uncertainty()[0]);
122  qA = quantile(a, 1-(alpha/2));
123  }else{
124  qA = e(A);
125  }
126  double qB;
127  if(B.uncertainty()[0]>0){
128  normal b(e(B), B.uncertainty()[0]);
129  qB = quantile(b, alpha/2);
130  }else{
131  qB = e(B);
132  }
133  //std::cout << " (quantiles " << qA << " vs. " << qB << ")";
134  if(qA >= qB){
135  return true;
136  }else{
137  return false;
138  }
139 
140  }

152  }else{
153  qA = e(A)[i];
154  }
155  double qB;
156  if(B.uncertainty()[i]>0){
157  normal b(e(B)[i], B.uncertainty()[i]);
158  qB = quantile(b, alpha/2);
159  }else{
160  qB = e(B)[i];
161  }
162  //std::cout << " (quantiles " << qA << " vs. " << qB << ")";
163  if(qA >= qB){
164  return true;
165  }
166  }
167  return false;
