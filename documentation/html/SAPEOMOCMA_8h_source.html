136  void updatePopulation(std::vector<IndividualType>& offspring, UncertainFunc_m const& function){
137  m_parents.insert(m_parents.end(),offspring.begin(),offspring.end());
138  m_selection(m_parents, function, mu(), this->strat, this->alpha);
139 
140  std::ofstream log;
141  log.open(this->file, std::ios::app);
142  log << std::endl;
143  for(unsigned int i=0; i<2*mu(); i++){
144  log << UncertainCMAIndividual<RealVector>::toString(m_parents[i]) << std::endl;
145  }
146  log.close();
147 
148  //determine from the selection which parent-offspring pair has been successful
149  for (std::size_t i = 0; i < mu(); i++) {
150  CMAChromosome::IndividualSuccess offspringSuccess = CMAChromosome::Unsuccessful;
151  //new update: an offspring is successful if it is selected
152  if (m_parents[mu()+i].selected()) {
153  m_parents[mu()+i].updateAsOffspring();
154  offspringSuccess = CMAChromosome::Successful;
155  }
156  if(m_parents[i].selected())
157  m_parents[i].updateAsParent(offspringSuccess);
158  }
159  //partition the selected individuals to the front and remove the unselected ones
160  std::partition(m_parents.begin(), m_parents.end(),IndividualType::IsSelected);
161  m_parents.erase(m_parents.begin()+mu(),m_parents.end());
162 
163  }
