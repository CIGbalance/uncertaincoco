83  void eval(double * res, const RealVector & p, bool forceCertain=false) const{
84  RealVector krigPoint;
85  if(this->surr==LOC || this->surr==BOTH){
86  for(unsigned int i=0; i<f1->numberOfVariables(); i++){
87  krigPoint.push_back(p(i));
88  }
89  }
90  if(this->surr==SURR || this->surr==BOTH){
91  RealVector surrEv = surrEval(p);
92  for(unsigned int i=0; i<f1->numberOfObjectives(); i++){
93  krigPoint.push_back(surrEv(i));
94  }
95  std::vector<double> pvec;
96  pvec.insert(pvec.end(), p.begin(), p.end());
97  krigPoints[pvec] = krigPoint;
98  }
99 
100  this->evalCounter++;
101  Archive arch = f1->getArchive();
102  bool uncertain = false;
103  bool fCertain = forceCertain;
104  bool notEvaluated = false;
105  std::vector<double> data;
106  if(arch.getNoPointsCurrent()< this->minK){
107  fCertain=true;
108  data.resize(3*this->f1->numberOfObjectives(), 0);
109  certainEval(res, p, krigPoint);
110  data.push_back(2);
111  }else{
112  ArchiveData arch_data = arch.getSample(std::min(this->k, arch.getNoPointsCurrent()), this->sampMode, krigPoint);
113  Kriging krig(arch_data, f1->numberOfVariables(), f1->numberOfObjectives(), -1);
114  std::vector<std::pair<double,double>> result = krig.predict(krigPoint);
115  std::vector<double> hist = Kriging::getHistory(krigPoint, this->h, std::min(this->k,
116  arch.getNoPointsCurrent()), arch, f1->numberOfObjectives(), this->sampMode);
117  for(unsigned int i=0; i< f1->numberOfObjectives(); i++){
118  data.push_back(result[i].first);
119  data.push_back(result[i].second);
120  data.push_back(hist[i]);
121  if(result[i].second>this->uncertaintyThreshold ||
122  hist[i]>this->historyThreshold){
123  uncertain=true;
124  }
125  }
126  if(uncertain){
127  certainEval(res, p, krigPoint);
128  data.push_back(1);
129  }else if(forceCertain){
130  certainEval(res, p, krigPoint);
131  data.push_back(2);
132  }else{
133  uncertainEval(res, &krig, p, krigPoint);
134  data.push_back(0);
135  }
136  }
137  notEvaluated = (data.back()==0); //This point was not evaluated
138  for(unsigned int i=0; i< f1->numberOfObjectives(); i++){
139  data.push_back(res[i]);
140  }
141  if(!uncertain && !fCertain){
142  double res2[f1->numberOfObjectives()];
143  cheaterEval(res2, p);
144  for(unsigned int i=0; i< f1->numberOfObjectives(); i++){
145  data.push_back(res2[i]);
146  }
147  }else{
148  for(unsigned int i=0; i< f1->numberOfObjectives(); i++){
149  data.push_back(res[i]);
150  }
151  }
152  if(f1->numberOfObjectives()==1){//Recommendation only currently implemented for SO
153  double newProgress = data.back();
154  if(!prevEvaluated){//previous individual was not evaluated
155  if(newProgress < progress){//Individual is better, replace prev
156  this->progress=newProgress;
157  this->points.back() = p;
158  }
159  //If not improved: do nothing
160  }else{//just add new individual
161  this->points.push_back(p);//TODO check if this actually a reference
162  this->progress = newProgress;
163 
164  }
165  prevEvaluated = !notEvaluated;//last individual not evaluated
166  }
167  /*std::cout << "data ";
168  for(unsigned int i=0; i<data.size(); ++i)
169  std::cout << data[i] << ' ';
170  std::cout << std::endl;*/
171  this->unArchive.addPoint(krigPoint, data, false);
172  return;
173  }
