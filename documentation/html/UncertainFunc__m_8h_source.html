58  virtual void uncertainEval(double * res, Kriging * krig, const RealVector & p, const RealVector krigPoint) const{
59  if(mode==MEAN){
60  std::vector<std::pair<double,double>> result = krig->predict(krigPoint);
61  for(unsigned int i=0; i<this->m_numberOfObjectives; i++){
62  res[i]=result[i].first;
63  }
64  }else if(mode==PoI){
65  std::vector<double> result = krig->probImprovement(krigPoint);
66  for(unsigned int i=0; i<this->m_numberOfObjectives; i++){
67  res[i]=(-1) *result[i];
68  }
69  }else if(mode==ExI){
70  std::vector<double> result = krig->expectedImprovement(krigPoint);
71  for(unsigned int i=0; i<this->m_numberOfObjectives; i++){
72  res[i]=result[i];
73  }
74  }else if(mode==LB){
75  std::vector<double> result = krig->lowerBound(krigPoint);
76  for(unsigned int i=0; i<this->m_numberOfObjectives; i++){
77  res[i]=(-1) * result[i];
78  }
79 
80  }else if(mode == CERTAIN){
81  double * result = COCOfunc<MultiObjectiveFunction>::eval(p, krigPoint);
82  for(unsigned int i=0; i<this->m_numberOfObjectives; i++){
83  res[i]=result[i];
84  }
85  delete[] result;
86  }
87 
88  }

143  void doInit(const COCOfunc_m * f1, int k, int h, double uncertaintyThreshold,
144  double historyThreshold, UncertainMode mode, SampleMode sampMode, int minK=-1){
145  if( (mode==PoI || mode==ExI) &&
146  (uncertaintyThreshold < std::numeric_limits<double>::infinity() ||
147  historyThreshold < std::numeric_limits<double>::infinity()) ){
148  throw std::invalid_argument("thresholds not implemented for improvement stuff");
149  }else if( mode == CERTAIN && (uncertaintyThreshold>0 || historyThreshold>0)){
150  throw std::invalid_argument("certain evaluation must have all thresholds as 0");
151  }
152  COCOfunc<MultiObjectiveFunction>::init(f1->getProblem(), f1->getCheaterProblem(),
153  f1->getTimeWindow(), f1->isNoisy(), 0, f1->getTransform());
154  if(this->m_numberOfObjectives==1){
155  throw std::invalid_argument("initialising multi-objective function with single-objective one");
156  }
157  this->mode = mode;
158  }
