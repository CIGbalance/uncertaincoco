108  virtual void init(UncertainFunc_m& function){
109  checkFeatures(function);
110  std::size_t dim = function.numberOfVariables();
111  RealVector lowerBounds(dim, -1E20);
112  RealVector upperBounds(dim, 1E20);
113  COCOfunc<MultiObjectiveFunction> * test = dynamic_cast<COCOfunc<MultiObjectiveFunction> *>(&function);
114  if(test!=NULL){
115  const double * lb = test->getLowerBounds();
116  const double * ub = test->getUpperBounds();
117  for(unsigned int i=0; i<dim; i++){
118  lowerBounds(i) = lb[i];
119  upperBounds(i) = ub[i];
120  }
121  }else{
122  if (function.hasConstraintHandler() && function.getConstraintHandler().isBoxConstrained()) {
123  typedef BoxConstraintHandler<RealVector> ConstraintHandler;
124  ConstraintHandler const& handler = static_cast<ConstraintHandler const&>(function.getConstraintHandler());
125 
126  lowerBounds = handler.lower();
127  upperBounds = handler.upper();
128  } else{
129  throw SHARKEXCEPTION("[SMS-EMOA::init] Algorithm does only allow box constraints");
130  }
131  }
132  doInit(lowerBounds, upperBounds,mu(),nm(),nc(),crossoverProbability(), function);

190  RealVector createOffspring(
191  std::vector<IndividualType>::const_iterator begin,
192  std::vector<IndividualType>::const_iterator end
193  )const{
194  TournamentSelection< IndividualType::RankOrdering > selection;
195 
196  IndividualType mate1( *selection(*mpe_rng, begin, end ) );
197  IndividualType mate2( *selection(*mpe_rng, begin, end) );
198 
199  if( coinToss(*mpe_rng, m_crossoverProbability ) ) {
200  m_crossover(*mpe_rng, mate1, mate2 );
201  }
202 
203  if( coinToss(*mpe_rng,0.5) ) {
204  m_mutator(*mpe_rng, mate1 );
205  return mate1.searchPoint();
206  } else {
207  m_mutator(*mpe_rng, mate2 );
208  return mate2.searchPoint();
209  }
210  }
