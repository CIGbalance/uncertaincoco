86  double * eval(RealVector point) const{
87  if(point.size()!=this->m_numberOfVariables){
88  throw std::invalid_argument("wrong input size");
89  }
90  double* x = point.storage();
91  double* y = new double[this->m_numberOfObjectives];
92  coco_evaluate_function(problem, x, &y[0]);
93  if(!this->inRegionOfInterest(point)){
94  for(unsigned int i=0; i<this->m_numberOfObjectives; i++){
95  y[i] +=this->m_penalty;
96  }
97  }
98  if(trans!=NO){
99  for(unsigned int i=0; i< this->m_numberOfObjectives; i++){
100  if(trans==LOG){
101  if(y[i]<=0){
102  throw std::invalid_argument("log transformation for negative values!");
103  }
104  y[i] = log(y[i]);
105  }else if(trans==INVERSE){
106  y[i] = -(1)/y[i];
107  }
108  }
109  }
110  std::vector<double> y_vec(y, y+this->m_numberOfObjectives);
111  this->archive.addPoint(point, y_vec);
112  return y;
113  }

197  ArchiveData lhs(int point_num, int seed, int duplication=5)const{
198  int *x;// An I4MAT is an MxN array of I4's, stored by (I,J) -> [I+J*M].
199  //m = dim_num, n = point_num
200  x = ihs ( m_numberOfVariables, point_num, duplication, seed);
201  //i4mat_transpose_print ( m_numberOfVariables, point_num, x, " X:" );
202  std::vector<RealVector> inputs(point_num);
203  std::vector<RealVector> labels(point_num);
204  for(int j=0; j<point_num; j++){
205  RealVector point(m_numberOfVariables);
206  for(unsigned int i=0; i<m_numberOfVariables; i++){
207  point(i) = x[i+j*m_numberOfVariables]-1;
208  point(i) = fabs(m_max[i]-m_min[i])*point(i)/(double)(point_num-1) + m_min[i]; //Rescaling
209  }
210  inputs[j] = point;
211  double * y = COCOfunc::eval(point);
212  RealVector values(m_numberOfObjectives);
213  for(unsigned int i=0; i<m_numberOfObjectives; i++){
214  values(i) = y[i];
215  }
216  labels[j] = values;
217  delete[] y;
218  }
219  delete[] x;
220  ArchiveData sample = createLabeledDataFromRange(inputs, labels);
221  return sample;
222  }
