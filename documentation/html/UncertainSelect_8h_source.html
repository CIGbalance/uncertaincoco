187  std::pair<unsigned int,std::size_t> selectionStep(std::vector<Individual>& population, std::size_t mu){
188  std::map<int, Pop> fronts;
189 
190  unsigned int maxRank=0;
191  for(unsigned int i=0; i< population.size(); i++){
192  if(fronts.find(population[i].rank())==fronts.end()){
193  fronts[population[i].rank()] = Pop();
194  if(population[i].rank()>maxRank){
195  maxRank = population[i].rank();
196  }
197  }
198  fronts[population[i].rank()].push_back(&population[i]);
199  population[i].selected() = true;
200  }
201 
202  //std::cout << "max Rank " << maxRank << std::endl;
203  //deselect the highest rank fronts until we would end up with less or equal mu elements
204  unsigned int rank = maxRank;
205  std::size_t popSize = population.size();
206  while(popSize-fronts[rank].size() >= mu){
207  //deselect all elements in this front
208  Pop front = fronts[rank];
209  for(std::size_t i = 0; i != front.size(); ++i){
210  front[i]->selected() = false;
211  //std::cout << UncertainIndividual<FitnessType>::toString(*front[i]) << " " << front[i]->rank() << std::endl;
212  }
213  popSize -= front.size();
214  --rank;
215  }
216  //delete &fronts;
217  //delete &population;
218  return std::make_pair(rank, popSize);
219  }
